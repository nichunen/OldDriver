class Solution {
    public boolean isHappy(int n) {
        Set<Integer> sumSet = new HashSet();

        while (!sumSet.contains(n)) {
            sumSet.add(n);
            int newn = 0;

            while (n > 0) {
                int single = n % 10;
                newn += (single * single);
                n = n / 10;
            }

            if (newn == 1) {
                return true;
            }

            n = newn;
        }

        return false;
    }
}
