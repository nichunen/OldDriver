class Solution {
    public int countPrimes(int n) {
        boolean[] primes = new boolean[n];

        int end = n % 2 == 0 ? n / 2 : (n + 2) / 2;

        for (int i = 2; i <= end; i++) {
            for (int j = 2; j <= end; j++) {
                int num = i * j;
                if (num > n) {
                    break;
                }
                primes[num - 1] = true;
            }
        }

        int result = 0;

        for (int i = 1; i < n - 1; i++) {
            if (!primes[i]) {
                result++;
            }
        }

        return result;
    }
}
